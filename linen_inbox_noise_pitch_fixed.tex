  \vspace{0.4cm}
  \begin{itemize}\setlength\itemsep{1.1em}
    \item[\textcolor{terra}{1.}] \textbf{The team feels swamped.} Most of that ``busy'' feeling

% ---- Slide: what is a label -----------------------------------------
\begin{frame}{First --- what is a ``label''?}
  \vspace{0.3cm}
  \begin{block}{The simple idea}
    \centering\large
    A label is just a \textcolor{terra}{\textbf{coloured sticker}} you put on an email.
  \end{block}
  \vspace{0.8em}
  \begin{itemize}\setlength\itemsep{0.7em}
    \item[\textcolor{teal}{$\bullet$}] \textbf{It sorts your mail for you.} Put an ``AusPost'' sticker on every
          delivery update, and you can see all of them together in one click.
    \item[\textcolor{teal}{$\bullet$}] \textbf{It tidies up without throwing anything away.} You can tell Gmail
          ``add the sticker and keep it out of my inbox.'' The email is filed
          safely away --- never deleted, and always there if you search for it.
  \end{itemize}
  \vspace{0.5em}
  \pause
  \begin{center}\small\textcolor{muted}{Like sorting post into trays --- nothing is binned, it's just put in the right place.}\end{center}
\end{frame}

% ---- Slide: the fix -------------------------------------------------
\begin{frame}{The fix: let Gmail sort the mail for us}
  \vspace{0.3cm}
  \begin{block}{One simple rule, run automatically on every email}
    \centering\large
    \textcolor{teal}{\textbf{Recognise the sender}} \;$\rightarrow$\;
    \textcolor{terra}{\textbf{Add a sticker + tuck it away}}
  \end{block}
  \vspace{0.8em}
  \begin{itemize}\setlength\itemsep{0.7em}
    \item Robot emails get \textbf{labelled and filed away} --- out of sight, still easy to find.
    \item Real customers \textbf{stay in the inbox} (and get handed to the AI).
    \item \textbf{Nothing is deleted, nothing is replied to.} The rule only tidies up.
  \end{itemize}
\end{frame}

% ---- Slide: the one rule we must get right --------------------------
\begin{frame}{The one thing we must be careful about}
  \vspace{0.4cm}
  \begin{block}{Don't accidentally hide website chat messages}
    When a customer types in the website chat, Shopify forwards it to
    \texttt{orders@} as an email. It \emph{looks} like a robot email, but it is a
    \textbf{real person} waiting for help.
  \end{block}
  \vspace{0.8em}
  \begin{itemize}\setlength\itemsep{0.6em}
    \item[\textcolor{teal}{\checkmark}] Tidy away Shopify \textbf{order / payout / system} emails.
    \item[\textcolor{terra}{$\times$}] Never make a lazy ``hide everything from Shopify'' rule.
    \item[\textcolor{teal}{\checkmark}] Give chat messages \textbf{their own sticker} and keep them in the inbox.
  \end{itemize}
\end{frame}

% ---- Slide: safe rollout --------------------------------------------
\begin{frame}{How we turn it on safely}
  \vspace{0.3cm}
  \begin{enumerate}\setlength\itemsep{0.8em}
    \item \textbf{Stickers only, at first.} Add the labels but leave every email in the inbox for a few days.
    \pause
    \item \textbf{Check nothing important got mislabelled.} Look through each sticker and confirm no real customer landed in the wrong tray.
    \pause
    \item \textbf{Then start tucking away.} Only once a sticker is proven safe do we let those emails skip the inbox.
    \pause
    \item \textbf{Never delete.} Everything stays searchable forever, in case of a dispute.
  \end{enumerate}
  \vspace{0.6em}
  \pause
  \begin{center}\small\textcolor{muted}{We can undo any step at any time.}\end{center}
\end{frame}

% ---- Slide: what the n8n workflow is (plain language) ---------------
\begin{frame}{What the ``AI assistant'' (n8n) actually does}
  \vspace{0.25cm}
  \begin{block}{Think of it as a tireless junior assistant}
    \small
    n8n is a tool that does a set of steps automatically, like a recipe. Ours
    reads each real customer email and gets it ready --- but a \textbf{human always
    presses send.}
  \end{block}
  \vspace{0.5em}
  \small
  \begin{enumerate}\setlength\itemsep{0.35em}
    \item \textbf{Watches} the inbox for a new customer email.
    \item \textbf{Ignores} obvious junk (auto-replies, bounced mail).
    \item \textbf{Checks} it hasn't already handled this one (no double-ups).
    \item \textbf{Reads \& sorts it} with AI: can I safely draft a reply, does a
          human need to handle this, or is no reply needed?
    \item If safe: \textbf{writes a draft} reply in Sweta's voice --- and \emph{stops}.
    \item \textbf{Adds a sticker} and notes it in a logbook (a Google Sheet).
  \end{enumerate}
\end{frame}
