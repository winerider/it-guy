% ---- Slide: the fix -------------------------------------------------
\begin{frame}{The fix: Gmail filters}
  \vspace{0.3cm}
  \begin{block}{One simple rule, applied automatically to every email}
    \centering\large
    \textcolor{teal}{\textbf{Match the sender}} \;$\rightarrow$\;
    \textcolor{terra}{\textbf{Label it + skip the inbox}}
  \end{block}
  \vspace{0.8em}
  \begin{itemize}\setlength\itemsep{0.7em}
    \item Noise is \textbf{labelled and archived} --- out of view, still fully searchable.
    \item Real customers \textbf{stay in the inbox} (and feed the AI).
    \item \textbf{Nothing is deleted}, nothing is replied to --- filters only organise.
  \end{itemize}
\end{frame}

% ---- Slide: the one rule we must get right --------------------------
\begin{frame}{The one thing we must get right}
  \vspace{0.4cm}
  \begin{block}{Do not bury Shopify chat relays}
    When a customer uses the website chat, Shopify relays it into \texttt{orders@}
    as an email. It \emph{looks} system-generated, but it's a \textbf{real customer}.
  \end{block}
  \vspace{0.8em}
  \begin{itemize}\setlength\itemsep{0.6em}
    \item[\textcolor{teal}{\checkmark}] Filter Shopify \textbf{order / payout / system} mail $\rightarrow$ archive.
    \item[\textcolor{terra}{$\times$}] Never write a blanket ``from: shopify.com $\rightarrow$ archive'' rule.
    \item[\textcolor{teal}{\checkmark}] Give chat relays their \textbf{own label} and keep them in the inbox.
  \end{itemize}
\end{frame}

% ---- Slide: safe rollout --------------------------------------------
\begin{frame}{How we roll it out safely}
  \vspace{0.3cm}
  \begin{enumerate}\setlength\itemsep{0.8em}
    \item \textbf{Label only first.} Apply labels but keep everything in the inbox for a few days.
    \pause
    \item \textbf{Verify the catch.} Apply retroactively and skim each label --- confirm no real customers landed in ``Notifications''.
    \pause
    \item \textbf{Then archive.} Turn on ``skip inbox + mark read'' only once a label has proven clean.
    \pause
    \item \textbf{Never delete.} Archive keeps everything searchable for disputes.
  \end{enumerate}
  \vspace{0.6em}
  \pause
  \begin{center}\small\textcolor{muted}{Fully reversible at every step.}\end{center}
\end{frame}
